    // ORDERS COLLECTION - Comprehensive rules for all order operations
    // ===========================================
    match /orders/{orderId} {
      
      // READ RULES - Allow reading orders based on different scenarios
      allow read: if request.auth != null && (
        // Allow drivers to read pending/confirmed/active orders for discovery
        resource.data.status in ['pending', 'confirmed', 'active'] ||
        // Allow assigned driver to read their orders
        resource.data.driverId == request.auth.uid ||
        // Allow order owner to read their orders (supporting both uid and userId fields)
        resource.data.uid == request.auth.uid ||
        resource.data.userId == request.auth.uid ||
        // Allow admins to read all orders
        request.auth.token.admin == true
      );
      
      // READ RULES - Allow all authenticated users to read orders (for testing)
      allow read: if request.auth != null;
      
      // LIST RULES - Allow users to query their own orders with where clauses
      allow list: if request.auth != null && (
        // Always allow listing for authenticated users with reasonable limits
        request.query.limit <= 50 ||
        // Allow unlimited listing for specific conditions
        request.auth.uid != null ||
        // Allow admins to list all orders
        request.auth.token.admin == true
      );
      
      // TEMPORARY: Allow any authenticated user to list orders for dashboard (remove this in production)
      allow list: if request.auth != null;
      
      // CREATE RULES - Comprehensive order creation rules
      allow create: if request.auth != null && (
        // Standard order creation
        (
          (request.auth.uid == request.resource.data.uid || 
           request.auth.uid == request.resource.data.userId) &&
          request.resource.data.keys().hasAll(['status', 'createdAt']) &&
          request.resource.data.status in ['pending', 'confirmed', 'active', 'pending_payment']
        ) ||
        // Shifting service orders
        (
          request.auth.uid == request.resource.data.uid &&
          request.resource.data.orderType == 'shifting' &&
          request.resource.data.status in ['pending', 'confirmed', 'active', 'pending_payment']
        ) ||
        // Pickup & drop orders
        (
          request.auth.uid == request.resource.data.uid &&
          request.resource.data.orderType == 'pickndrop' &&
          request.resource.data.status == 'confirmed'
        )
      );
      
      // UPDATE RULES - Complex logic for different update scenarios
      allow update: if request.auth != null && (
        
        // SCENARIO 1: Driver accepting an order
        (
          resource.data.status in ['pending', 'confirmed', 'active'] &&
          request.resource.data.status == 'accepted' &&
          request.resource.data.driverId == request.auth.uid &&
          request.resource.data.keys().hasAll(['acceptedAt', 'driverLocation']) &&
          request.resource.data.uid == resource.data.uid &&
          request.resource.data.createdAt == resource.data.createdAt
        ) ||
        
        // SCENARIO 2: Driver rejecting an order (adding to rejection list)
        (
          resource.data.status in ['pending', 'confirmed', 'active'] &&
          request.resource.data.keys().hasAny(['rejectedBy']) &&
          request.resource.data.uid == resource.data.uid &&
          request.resource.data.status == resource.data.status &&
          request.resource.data.createdAt == resource.data.createdAt
        ) ||
        
        // SCENARIO 3: Order owner updating their own order
        (
          (resource.data.uid == request.auth.uid || 
           resource.data.userId == request.auth.uid) &&
          (
            !resource.data.keys().hasAny(['driverId']) ||
            resource.data.driverId == request.resource.data.driverId
          )
        ) ||
        
        // SCENARIO 4: Assigned driver updating order progress
        (
          resource.data.driverId == request.auth.uid &&
          request.resource.data.uid == resource.data.uid &&
          request.resource.data.createdAt == resource.data.createdAt
        ) ||
        
        // SCENARIO 5: Payment status updates for shifting/pickup orders
        (
          resource.data.uid == request.auth.uid &&
          request.resource.data.uid == request.auth.uid &&
          (
            request.resource.data.status in ['confirmed', 'paid'] ||
            request.resource.data.keys().hasAny(['paymentMethod', 'paymentStatus', 'paymentDetails'])
          )
        )
      );
      
      // DELETE RULES - Only order owners and admins
      allow delete: if request.auth != null && (
        resource.data.uid == request.auth.uid ||
        resource.data.userId == request.auth.uid ||
        request.auth.token.admin == true
      );
      
      // UPDATE RULES - Allow admins to update any order
      allow update: if request.auth != null && request.auth.token.admin == true;
    }

    // Feature flags for app functionality
    match /featureFlags/{flagId} {
      allow read: if request.auth != null;
      allow write: if request.auth != null && 
        request.auth.token.admin == true;
    }
    
    // App version and update information
    match /appVersion/{versionId} {
      allow read: if request.auth != null;
      allow write: if request.auth != null && 
        request.auth.token.admin == true;
    }
    
    // Service availability status
    match /serviceStatus/{serviceId} {
      allow read: if request.auth != null;
      allow create, update: if request.auth != null;
      allow write: if request.auth != null && 
        request.auth.token.admin == true;
    }
    
    // ===========================================
    // ADDITIONAL SYSTEM COLLECTIONS
    // ===========================================
    
    // Emergency notifications and alerts
    match /emergencyAlerts/{alertId} {
      allow read: if request.auth != null;
      allow create, update, delete: if request.auth != null && 
        request.auth.token.admin == true;
    }
    
    // System configuration and flags
    match /systemConfig/{configId} {
      allow read: if request.auth != null;
      allow write: if request.auth != null && 
        request.auth.token.admin == true;
    }
    
    // API rate limiting and throttling
    match /rateLimits/{limitId} {
      allow read: if request.auth != null;
      allow write: if request.auth != null && 
        request.auth.token.admin == true;
    }
    
    // Health check and monitoring
    match /healthCheck/{checkId} {
      allow read: if request.auth != null;
      allow create, update: if request.auth != null;
      allow write: if request.auth != null && 
        request.auth.token.admin == true;
    }
    
    // ===========================================
    // ERROR HANDLING AND FALLBACK RULES
    // ===========================================
    
    // Generic fallback for user's own data
    match /{collection}/{documentId} {
      allow read, write: if request.auth != null && 
        collection in ['userSettings', 'userHistory', 'userFavorites'] &&
        documentId == request.auth.uid;
    }
    
    // Fallback rule for system collections
    match /{collection}/{documentId} {
      allow read: if request.auth != null && 
        collection in [
          'maintenance', 'systemStatus', 'appStatus', 'featureFlags', 
          'appVersion', 'serviceStatus', 'emergencyAlerts', 'systemConfig',
          'rateLimits', 'healthCheck'
        ];
      
      allow write: if request.auth != null && 
        request.auth.token.admin == true &&
        collection in [
          'maintenance', 'systemStatus', 'appStatus', 'featureFlags', 
          'appVersion', 'serviceStatus', 'emergencyAlerts', 'systemConfig',
          'rateLimits', 'healthCheck'
        ];
    }
  }
